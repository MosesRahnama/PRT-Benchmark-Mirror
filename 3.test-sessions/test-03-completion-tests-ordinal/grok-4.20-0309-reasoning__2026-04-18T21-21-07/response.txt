### 1. `R_rec_succ` case (recursive successor step)
```lean
| _, _, Step.R_rec_succ b s n => by
    simp [mu]
    -- mu(recDelta b s (delta n)) = ω^(μ(δn) + μs + 6) + ω·(μb+1) + 1
    -- μ(δn) = ω^5·(μn+1) + 1, so the exponent already contains an ω^5 factor.
    -- The RHS (app) is a polynomial of degree < 3 in terms whose largest
    -- power is μ(recDelta b s n) < ω^(μn + μs + 6). This is dominated by
    -- any ω raised to an exponent ≥ μn + μs + 6 + 1.
    have h_exp : mu n + mu s + 6 < mu (delta n) + mu s + 6 := by
      simp [mu]; exact Ordinal.add_lt_add_right (Ordinal.lt_add_of_pos_right one_pos) _
    have h_rec_dom : mu (recDelta b s n) < omega0 ^ (mu n + mu s + 6) :=
      Ordinal.lt_add_of_pos_right (Ordinal.opow_pos _ Ordinal.one_pos)
    calc
      mu (app s (recDelta b s n))
        < omega0 ^ 3 * (mu s + 1) + omega0 ^ 3 * (mu (recDelta b s n) + 1) :=
            by simp only [add_assoc, add_comm]; apply add_lt_add_right; exact lt_add_of_le_of_lt (le_add_right ..) (Ordinal.opow_lt_opow_left h_rec_dom (by decide))
      _ < omega0 ^ (mu n + mu s + 6) :=
            by apply Ordinal.add_lt_opow; · exact Ordinal.opow_pos _ Ordinal.one_pos
               · exact lt_of_le_of_lt (Ordinal.add_le_add (le_opow_of_base_le_one ..) (Ordinal.mul_le_opow ..)) h_exp
      _ ≤ mu (recDelta b s (delta n)) := by simp [mu]; exact le_add_right _ _
```

### 3. `R_eq_diff` case (distinct equality)
```lean
| _, _, Step.R_eq_diff a b => by
    simp [mu]
    -- μ(integrate(merge a b)) expands to a term whose Cantor normal form
    -- has leading exponent 7 (ω^4·(ω^3·(μa+1) + …) ≈ ω^7·…). The redex
    -- side is ω^(μa + μb + 9). Because μa + μb + 9 ≥ 9 > 7, the inequality
    -- follows from monotonicity of ω^·.
    have h_merge : mu (merge a b) < omega0 ^ (mu a + mu b + 4) := by
      simp [mu]
      apply Ordinal.add_lt_opow _ Ordinal.one_pos
      apply Ordinal.add_lt_opow _ Ordinal.one_pos
      · exact lt_add_of_le_of_lt (le_mul_right ..) (Ordinal.opow_pos _ Ordinal.one_pos)
      · exact lt_add_of_le_of_lt (le_mul_right ..) (Ordinal.opow_pos _ Ordinal.one_pos)
    calc
      mu (integrate (merge a b))
        = omega0 ^ 4 * (mu (merge a b) + 1) + 1 := rfl
      _ < omega0 ^ 4 * omega0 ^ (mu a + mu b + 4) := by
            apply Ordinal.add_lt_add_left; apply Ordinal.mul_lt_mul_of_pos_left
            · exact Ordinal.lt_add_one _
            · exact Ordinal.opow_pos _ Ordinal.one_pos
      _ = omega0 ^ (mu a + mu b + 8) := by rw [Ordinal.opow_add]; rfl
      _ < omega0 ^ (mu a + mu b + 9) := Ordinal.opow_lt_opow_left (by simp_arith) (Ordinal.add_lt_add_right (by decide) _)
      _ ≤ mu (eqW a b) := by simp [mu]; exact le_add_right _ _
```
