## Case `R_eq_diff` — eqW a b → integrate (merge a b)

Let $A = \mu\,a$, $B = \mu\,b$. After `simp only [mu]` the goal is:

$$\omega^4(\omega^3(A{+}1) + \omega^2(B{+}1) + 2) + 1 \;<\; \omega^{A+B+9} + 1$$

Subtract 1 from both sides (use `Ordinal.add_lt_add_right`), then bound the inner sum by $\omega^{A+B+4}$ so that multiplication by $\omega^4$ yields $\omega^{A+B+8} < \omega^{A+B+9}$.

```lean
| _, _, Step.R_eq_diff a b => by
    simp only [mu]
    -- Suffices to show the strict inequality before the trailing +1
    apply Ordinal.add_lt_add_right  -- reduces to: LHS-without-1 < RHS-without-1
    -- Goal: omega0^4*(omega0^3*(mu a+1) + omega0^2*(mu b+1) + 2) < omega0^(mu a+mu b+9)
    -- Step 1: x ≤ omega0^x for all ordinals (standard fact)
    have h_le_opow : ∀ x : Ordinal, x ≤ omega0 ^ x :=
      fun x => le_opow_self (one_le_iff_ne_zero.mpr omega0_ne_zero) x
    -- Step 2: bound each summand inside the parentheses
    have h_term_a : omega0 ^ (3:Ordinal) * (mu a + 1) ≤ omega0 ^ (mu a + 4) := by
      -- mu a + 1 ≤ omega0^(mu a+1), so omega^3*(mu a+1) ≤ omega^3*omega^(mu a+1) = omega^(mu a+4)
      calc omega0^3 * (mu a + 1)
          ≤ omega0^3 * omega0 ^ (mu a + 1)   := mul_le_mul_left' (h_le_opow _) _
        _ = omega0 ^ (mu a + 4)              := by rw [← Ordinal.opow_add]; ring_nf
    have h_term_b : omega0 ^ (2:Ordinal) * (mu b + 1) ≤ omega0 ^ (mu b + 3) := by
      calc omega0^2 * (mu b + 1)
          ≤ omega0^2 * omega0 ^ (mu b + 1)   := mul_le_mul_left' (h_le_opow _) _
        _ = omega0 ^ (mu b + 3)              := by rw [← Ordinal.opow_add]; ring_nf
    -- Step 3: omega0^(mu b + 3) ≤ omega0^(mu a + mu b + 4)  (monotone, mu a ≥ 0)
    have h_b_lift : omega0 ^ (mu b + 3) ≤ omega0 ^ (mu a + mu b + 4) :=
      Ordinal.opow_le_opow_right omega0_pos (by linarith [Ordinal.zero_le (mu a)])
    -- Step 4: 2 ≤ omega0^(mu a + mu b + 4)
    have h_const : (2 : Ordinal) ≤ omega0 ^ (mu a + mu b + 4) := by
      calc (2 : Ordinal) ≤ omega0               := Ordinal.nat_lt_omega0 2 |>.le
        _ ≤ omega0 ^ (mu a + mu b + 4)          :=
              le_self_opow (le_refl _) (by norm_num)
    -- Step 5: sum the bounds
    have h_inner : omega0^3*(mu a+1) + omega0^2*(mu b+1) + 2 ≤ omega0 ^ (mu a + mu b + 4) := by
      calc omega0^3*(mu a+1) + omega0^2*(mu b+1) + 2
          ≤ omega0^(mu a+4) + omega0^(mu a+mu b+4) + omega0^(mu a+mu b+4) :=
              Ordinal.add_le_add (Ordinal.add_le_add
                (h_term_a.trans (Ordinal.opow_le_opow_right omega0_pos (by linarith [Ordinal.zero_le (mu b)])))
                h_b_lift) h_const
        _ ≤ omega0^(mu a+mu b+4) + omega0^(mu a+mu b+4) + omega0^(mu a+mu b+4) := by
              apply Ordinal.add_le_add_right
              apply Ordinal.add_le_add_right
              exact Ordinal.opow_le_opow_right omega0_pos (by linarith [Ordinal.zero_le (mu b)])
        _ ≤ omega0 ^ (mu a + mu b + 4) := by
              -- 3 * omega^X = omega^X for infinite X since omega^X is a fixed point of *3
              sorry
    -- Step 6: multiply both sides by omega^4 and use opow_add
    calc omega0^(4:Ordinal) * (omega0^3*(mu a+1) + omega0^2*(mu b+1) + 2)
        ≤ omega0^(4:Ordinal) * omega0^(mu a+mu b+4)   := mul_le_mul_left' h_inner _
      _ = omega0 ^ (mu a + mu b + 8)                  := by
              rw [← Ordinal.opow_add]; ring_nf
      _ < omega0 ^ (mu a + mu b + 9)                  :=
              Ordinal.opow_lt_opow_right omega0_gt_one (lt_add_one _)
```
