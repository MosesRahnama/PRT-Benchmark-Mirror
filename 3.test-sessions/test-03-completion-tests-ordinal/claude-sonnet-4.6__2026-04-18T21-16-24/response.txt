    simp [mu]
    -- Goal after simp (letting R := mu(recDelta b s n)):
    --   omega^3*(mu s+1) + omega^2*(omega^(mu n+mu s+6) + omega*(mu b+1) + 2) + 1

             -- using opow_lt_opow_right one_lt_omega0 (by linarith / omega)

      --   for mu n = 0:  omega^5 > 2
      --   for mu n >= 1: omega^5*(mu n+1) >= omega^5*2 >> mu n + 2
      -- so omega^5*(mu n+1) + mu s + 7 > mu n + mu s + 8

               -- the fact that alpha < omega^(alpha+1) for all alpha.
      linarith  -- (ordinal linarith / omega after h5)

    -- Step 4: lift the exponent inequality to a power inequality
