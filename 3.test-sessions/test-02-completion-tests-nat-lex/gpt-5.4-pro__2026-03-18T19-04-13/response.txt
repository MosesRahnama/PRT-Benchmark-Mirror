In
, four of the five `sorry` branches can be sketched directly. The `R_rec_succ` branch is blocked by the current measure.

```lean
-- integrate (delta t) --> void
| _, _, Step.R_int_delta t => by
    by_cases hk0 : kappa t = 0
    · have hmu : mu void < mu (integrate (delta t)) := by
        simp [mu]
      have hk : kappa void = kappa (integrate (delta t)) := by
        simp [kappa, hk0]
      exact drop_right hmu hk
    · have hk : kappa void < kappa (integrate (delta t)) := by
        have hpos : 0 < kappa t := Nat.pos_of_ne_zero hk0
        simpa [kappa] using hpos
      exact drop_left hk

-- recDelta b s void --> b
| _, _, Step.R_rec_zero b s => by
    by_cases hsb : kappa s ≤ kappa b
    · have hmu : mu b < mu (recDelta b s void) := by
        simp [mu]
        omega
      have hk : kappa b = kappa (recDelta b s void) := by
        simp [kappa, Nat.max_eq_left hsb]
      exact drop_right hmu hk
    · have hlt : kappa b < kappa s := Nat.lt_of_not_ge hsb
      have hk : kappa b < kappa (recDelta b s void) := by
        have : kappa b < Nat.max (kappa b) (kappa s) :=
          lt_of_lt_of_le hlt (Nat.le_max_right _ _)
        simpa [kappa] using this
      exact drop_left hk

-- recDelta b s (delta n) --> app s (recDelta b s n)
| _, _, Step.R_rec_succ b s n => by
    -- This branch is not provable with the current `(kappa, mu)`.
    -- Counterexample: b = void, s = void, n = delta void.
    -- Then
    --   measure (recDelta void void (delta (delta void))) = (1, 3)
    --   measure (app void (recDelta void void (delta void))) = (1, 4)
    -- so the target lexicographic decrease fails.
    --
    -- Intended shape after repairing the measure/invariant:
    -- have hk :
    --   kappa (app s (recDelta b s n)) < kappa (recDelta b s (delta n)) := by
    --   ...
    -- exact drop_left hk
    sorry

-- eqW a a --> void
| _, _, Step.R_eq_refl a => by
    by_cases hk0 : kappa a = 0
    · have hmu : mu void < mu (eqW a a) := by
        simp [mu]
        omega
      have hk : kappa void = kappa (eqW a a) := by
        simp [kappa, hk0]
      exact drop_right hmu hk
    · have hk : kappa void < kappa (eqW a a) := by
        have hpos : 0 < kappa a := Nat.pos_of_ne_zero hk0
        simpa [kappa] using hpos
      exact drop_left hk

-- eqW a b --> integrate (merge a b)
| _, _, Step.R_eq_diff a b => by
    have hmu : mu (integrate (merge a b)) < mu (eqW a b) := by
      simp [mu]
      omega
    have hk : kappa (integrate (merge a b)) = kappa (eqW a b) := by
      simp [kappa]
    exact drop_right hmu hk
```

The key point: `R_rec_succ` needs a different measure or a stronger invariant on `n`. With the current definitions, `measure_decreases` is false in general. Not compiler-checked.
