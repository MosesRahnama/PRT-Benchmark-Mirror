A plausible replacement for the five `sorry`s is below. Four branches are routine. The `R_rec_succ` branch exposes a genuine obstruction with the current `(kappa, mu)` measure when `n = delta _`, so that one can only be sketched honestly.

```lean
-- integrate (delta t) --> void
| _, _, Step.R_int_delta t => by
    by_cases hk : kappa t = 0
    · have hmu : mu void < mu (integrate (delta t)) := by
        simp [mu]
      have hk' : kappa void = kappa (integrate (delta t)) := by
        simp [kappa, hk]
      exact drop_right hmu hk'
    · have hk' : kappa void < kappa (integrate (delta t)) := by
        simpa [kappa] using (Nat.pos_of_ne_zero hk : 0 < kappa t)
      exact drop_left hk'

-- recDelta b s void --> b
| _, _, Step.R_rec_zero b s => by
    have hle : kappa b ≤ kappa (recDelta b s void) := by
      dsimp [kappa]
      exact le_trans (Nat.le_max_left _ _) (Nat.le_max_left _ _)
    by_cases hk : kappa b = kappa (recDelta b s void)
    · have hmu : mu b < mu (recDelta b s void) := by
        simp [mu]
        omega
      exact drop_right hmu hk
    · have hk' : kappa b < kappa (recDelta b s void) := by
        omega
      exact drop_left hk'

-- recDelta b s (delta n) --> app s (recDelta b s n)
| _, _, Step.R_rec_succ b s n => by
    /-
    Skeleton:
      Let M := Nat.max (Nat.max (kappa b) (kappa s)) (kappa n).

      1. Show `kappa s ≤ M`.
      2. For every non-`delta` shape of `n`, prove
           kappa (recDelta b s n) = M
         and hence
           kappa (app s (recDelta b s n)) = M
         by `Nat.max_eq_right`.
      3. Since
           kappa (recDelta b s (delta n)) = M + 1,
         `drop_left` closes those subcases.

      The problematic subcase is `n = delta n'`:
        then
          kappa (app s (recDelta b s n))
            = kappa (recDelta b s (delta n)),
        while
          mu (app s (recDelta b s n))
            > mu (recDelta b s (delta n)).
      So the current lex measure cannot finish this branch as stated.
      This case needs either:
        * a strengthened first component of the measure, or
        * an extra side condition excluding `n = delta _`.
    -/
    cases n with
    | void =>
        have hs :
            kappa s ≤ Nat.max (Nat.max (kappa b) (kappa s)) (kappa void) := by
          exact le_trans (Nat.le_max_right _ _) (Nat.le_max_left _ _)
        have hk :
            kappa (app s (recDelta b s void)) <
            kappa (recDelta b s (delta void)) := by
          simp [kappa, Nat.max_eq_right hs]
        exact drop_left hk
    | delta n' =>
        sorry
    | integrate n' =>
        have hs :
            kappa s ≤ Nat.max (Nat.max (kappa b) (kappa s)) (kappa (integrate n')) := by
          exact le_trans (Nat.le_max_right _ _) (Nat.le_max_left _ _)
        have hk :
            kappa (app s (recDelta b s (integrate n'))) <
            kappa (recDelta b s (delta (integrate n'))) := by
          simp [kappa, Nat.max_eq_right hs]
        exact drop_left hk
    | merge n₁ n₂ =>
        have hs :
            kappa s ≤ Nat.max (Nat.max (kappa b) (kappa s)) (kappa (merge n₁ n₂)) := by
          exact le_trans (Nat.le_max_right _ _) (Nat.le_max_left _ _)
        have hk :
            kappa (app s (recDelta b s (merge n₁ n₂))) <
            kappa (recDelta b s (delta (merge n₁ n₂))) := by
          simp [kappa, Nat.max_eq_right hs]
        exact drop_left hk
    | app n₁ n₂ =>
        have hs :
            kappa s ≤ Nat.max (Nat.max (kappa b) (kappa s)) (kappa (app n₁ n₂)) := by
          exact le_trans (Nat.le_max_right _ _) (Nat.le_max_left _ _)
        have hk :
            kappa (app s (recDelta b s (app n₁ n₂))) <
            kappa (recDelta b s (delta (app n₁ n₂))) := by
          simp [kappa, Nat.max_eq_right hs]
        exact drop_left hk
    | recDelta b' s' n' =>
        have hs :
            kappa s ≤ Nat.max (Nat.max (kappa b) (kappa s)) (kappa (recDelta b' s' n')) := by
          exact le_trans (Nat.le_max_right _ _) (Nat.le_max_left _ _)
        have hk :
            kappa (app s (recDelta b s (recDelta b' s' n'))) <
            kappa (recDelta b s (delta (recDelta b' s' n'))) := by
          simp [kappa, Nat.max_eq_right hs]
        exact drop_left hk
    | eqW a b' =>
        have hs :
            kappa s ≤ Nat.max (Nat.max (kappa b) (kappa s)) (kappa (eqW a b')) := by
          exact le_trans (Nat.le_max_right _ _) (Nat.le_max_left _ _)
        have hk :
            kappa (app s (recDelta b s (eqW a b'))) <
            kappa (recDelta b s (delta (eqW a b'))) := by
          simp [kappa, Nat.max_eq_right hs]
        exact drop_left hk

-- eqW a a --> void
| _, _, Step.R_eq_refl a => by
    by_cases hk : kappa a = 0
    · have hmu : mu void < mu (eqW a a) := by
        simp [mu]
      have hk' : kappa void = kappa (eqW a a) := by
        simp [kappa, hk]
      exact drop_right hmu hk'
    · have hk' : kappa void < kappa (eqW a a) := by
        simpa [kappa] using (Nat.pos_of_ne_zero hk : 0 < kappa a)
      exact drop_left hk'

-- eqW a b --> integrate (merge a b)
| _, _, Step.R_eq_diff a b => by
    have hmu : mu (integrate (merge a b)) < mu (eqW a b) := by
      simp [mu]
      omega
    have hk : kappa (integrate (merge a b)) = kappa (eqW a b) := by
      simp [kappa]
    exact drop_right hmu hk
```

The only branch that does not look completable with the current definitions is `Step.R_rec_succ` when `n = delta _`. If needed, a next step is to redesign the first measure component so that peeling one outer `delta` in the third `recDelta` argument is always strictly decreasing.
