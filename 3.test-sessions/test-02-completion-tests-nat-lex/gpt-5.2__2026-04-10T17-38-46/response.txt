-- recDelta b s void --> b
| _, _, Step.R_rec_zero b s => by
    -- Always: mu(b) < mu(recDelta b s void)
    have hmu : mu b < mu (recDelta b s void) := by
      simp [mu]; omega

    -- For kappa: either it ties (then use mu), or it strictly drops (then use kappa).
    -- Split on which branch of the max controls kappa(recDelta b s void).
    cases Nat.le_total (kappa s) (kappa b) with
    | inl hs_le =>
        -- kappa(recDelta ...) = kappa b, so use drop_right via mu
        have hk : kappa b = kappa (recDelta b s void) := by
          -- simp should reduce kappa(recDelta b s void) to max (max kb ks) 0,
          -- then use hs_le to collapse max kb ks to kb.
          -- (You may need to hand-hold simp with Nat.max_eq_left / Nat.max_eq_right.)
          simp [kappa, hs_le, Nat.max_eq_left, Nat.max_eq_right, Nat.zero_le]
        exact drop_right hmu hk
    | inr hb_le =>
        -- Here kb ≤ ks; refine further into kb < ks (drop_left) or kb = ks (drop_right).
        cases lt_or_eq_of_le hb_le with
        | inl hk_lt =>
            have hk : kappa b < kappa (recDelta b s void) := by
              -- show kappa(recDelta ...) = kappa s, then rewrite to hk_lt
              -- (again: simp + Nat.max_eq_right + Nat.zero_le)
              simpa [kappa, hb_le, Nat.max_eq_left, Nat.max_eq_right, Nat.zero_le] using hk_lt
            exact drop_left hk
        | inr hk_eq =>
            have hk : kappa b = kappa (recDelta b s void) := by
              -- if kb = ks, then kappa(recDelta ...) collapses to kb
              simp [kappa, hk_eq, Nat.max_eq_left, Nat.max_eq_right, Nat.zero_le]
            exact drop_right hmu hk
