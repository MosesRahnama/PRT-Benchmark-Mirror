`drop_left` (strict κ-decrease) and `drop_right` (μ-decrease with κ-tie).

```lean
-- integrate (delta t) ⟶ void
| _, _, Step.R_int_delta t => by
    -- κ may drop strictly (when κ t > 0) or tie at 0.
    -- Either way we obtain a lexicographic decrease.
    have hμ  : mu void < mu (integrate (delta t)) := by simp [mu]
    have hκ₀ : kappa void = 0                       := by simp [kappa]
    have hκ₁ : kappa (integrate (delta t)) = kappa t := by simp [kappa]
    cases h : kappa t with
    | zero   =>
        -- κ-tie case: use μ-drop
        have hκeq : kappa void = kappa (integrate (delta t)) := by
          simpa [hκ₀,hκ₁,h] 
        exact drop_right hμ hκeq
    | succ n =>
        -- strict κ-drop case
        have hκlt : kappa void < kappa (integrate (delta t)) := by
          simpa [hκ₀,hκ₁,h] using Nat.zero_lt_succ _
        exact drop_left hκlt

-- recDelta b s void ⟶ b
| _, _, Step.R_rec_zero b s => by
    -- κ can tie (when κ b is maximal) or drop; μ always drops.
    have hμ  : mu b < mu (recDelta b s void) := by simp [mu]
    have hκle : kappa b ≤ kappa (recDelta b s void) := by
      simp [kappa, Nat.le_max_left]
    cases lt_or_eq_of_le hκle with
    | inl hκlt =>
        exact drop_left hκlt
    | inr hκeq =>
        exact drop_right hμ hκeq

-- recDelta b s (delta n) ⟶ app s (recDelta b s n)
| _, _, Step.R_rec_succ b s n => by
    -- κ drops strictly by exactly 1.
    have hκlt : kappa (app s (recDelta b s n))
                 < kappa (recDelta b s (delta n)) := by
      simp [kappa, Nat.max_assoc, Nat.max_left_comm, Nat.lt_add_of_pos_right]
    exact drop_left hκlt

-- eqW a a ⟶ void
| _, _, Step.R_eq_refl a => by
    have hμ  : mu void < mu (eqW a a) := by simp [mu]
    have hκ₀ : kappa void = 0         := by simp [kappa]
    have hκ₁ : kappa (eqW a a) = kappa a := by simp [kappa]
    cases h : kappa a with
    | zero   =>
        have hκeq : kappa void = kappa (eqW a a) := by
          simpa [hκ₀,hκ₁,h]
        exact drop_right hμ hκeq
    | succ n =>
        have hκlt : kappa void < kappa (eqW a a) := by
          simpa [hκ₀,hκ₁,h] using Nat.zero_lt_succ _
        exact drop_left hκlt

-- eqW a b ⟶ integrate (merge a b)
| _, _, Step.R_eq_diff a b => by
    -- κ ties; μ decreases by 1.
    have hμ : mu (integrate (merge a b)) < mu (eqW a b) := by
      simp [mu]
    have hκ : kappa (integrate (merge a b)) = kappa (eqW a b) := by
      simp [kappa]
    exact drop_right hμ hκ
```

The proof pattern is uniform:

1.  Compute κ/μ for source and target with `simp`.
2.  Identify whether κ decreases strictly (`drop_left`) or remains equal (`drop_right`).
3.  Supply the corresponding inequality/equality proofs.
