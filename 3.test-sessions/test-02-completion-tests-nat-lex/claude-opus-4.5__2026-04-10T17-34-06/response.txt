-- integrate (delta t) --> void
-- kappa: kappa(integrate (delta t)) = kappa t, kappa(void) = 0
-- mu: mu(integrate (delta t)) = 1 + mu t, mu(void) = 0
-- Strategy: If kappa t > 0, kappa drops. Otherwise kappa ties at 0 and mu drops.
| _, _, Step.R_int_delta t => by
    by_cases h : kappa t = 0
    · -- kappa ties at 0; mu drops from (1 + mu t) to 0
      have hmu : mu void < mu (integrate (delta t)) := by simp [mu]; omega
      have hk  : kappa void = kappa (integrate (delta t)) := by simp [kappa, h]
      exact drop_right hmu hk
    · -- kappa drops from kappa t to 0
      have hk : kappa void < kappa (integrate (delta t)) := by
        simp [kappa]; omega
      exact drop_left hk

-- recDelta b s void --> b
-- kappa: kappa(LHS) = max(max(kappa b, kappa s), 0) = max(kappa b, kappa s)
-- kappa(RHS) = kappa b ≤ max(kappa b, kappa s)
-- mu: mu(LHS) = 3 + mu b + mu s + 0, mu(RHS) = mu b
-- Strategy: If kappa s > kappa b, kappa drops. Otherwise kappa ties and mu drops.
| _, _, Step.R_rec_zero b s => by
    by_cases h : kappa s ≤ kappa b
    · -- kappa ties; mu drops from (3 + mu b + mu s) to mu b
      have hmu : mu b < mu (recDelta b s void) := by simp [mu]; omega
      have hk  : kappa b = kappa (recDelta b s void) := by
        simp [kappa, Nat.max_eq_left h]
      exact drop_right hmu hk
    · -- kappa drops from max(kappa b, kappa s) to kappa b
      have hk : kappa b < kappa (recDelta b s void) := by
        simp [kappa]; omega
      exact drop_left hk

-- kappa(LHS) = max(max(kappa b, kappa s), kappa n) + 1 (delta pattern adds +1)
-- Key insight: The +1 bonus on LHS is removed when unwrapping delta
-- Case analysis on whether n is itself a delta term:
--   - If n is not delta-shaped: kappa(recDelta b s n) has no +1, so kappa strictly drops
--   - If n = delta m: Need more careful analysis (nested delta peeling)
| _, _, Step.R_rec_succ b s n => by

    -- After stepping, the inner recDelta's kappa depends on n's structure

    -- When n is not delta: ε = 0, so kappa drops by at least 1

    --   showing termination via the total number of delta wrappers

    | delta m =>
        -- n = delta m: inner recDelta also gets +1 bonus

        sorry
    | _ =>

        have hk : kappa (app s (recDelta b s _)) < kappa (recDelta b s (delta _)) := by
          simp [kappa]
          -- max(ks, max(max kb ks, kn)) < max(max kb ks, kn) + 1
          omega
        exact drop_left hk

-- eqW a a --> void
-- kappa: kappa(eqW a a) = max(kappa a, kappa a) = kappa a, kappa(void) = 0
-- mu: mu(eqW a a) = 4 + 2*mu a, mu(void) = 0
-- Strategy: If kappa a > 0, kappa drops. Otherwise kappa ties at 0 and mu drops.
| _, _, Step.R_eq_refl a => by

      exact drop_left hk

-- eqW a b --> integrate (merge a b)
-- kappa: Both sides have kappa = max(kappa a, kappa b), so kappa ties
-- mu: mu(eqW a b) = 4 + mu a + mu b
--     mu(integrate (merge a b)) = 1 + 2 + mu a + mu b = 3 + mu a + mu b

    have hk : kappa (integrate (merge a b)) = kappa (eqW a b) := by

    exact drop_right hmu hk

**Summary of proof strategies:**

| Rule | Approach |
|------|----------|
| `R_int_delta` | Case split on `kappa t = 0`. If 0, kappa ties and mu drops. Otherwise kappa drops. |
| `R_rec_zero` | Case split on `kappa s ≤ kappa b`. If true, kappa ties (both = kappa b) and mu drops. Otherwise kappa drops. |
| `R_rec_succ` | Case split on whether `n` is a delta constructor. Non-delta case: kappa drops by 1 (the +1 bonus disappears). Delta case: **requires additional reasoning**—the given measure may not suffice; consider a refined measure counting nested delta depth. |
| `R_eq_refl` | Case split on `kappa a = 0`. Same pattern as `R_int_delta`. |
| `R_eq_diff` | Direct: kappa ties trivially, mu drops from 4 to 3. |

**Note on `R_rec_succ`:** The nested-delta subcase (when `n = delta m`) is the hardest. The kappa measure ties and mu *increases*. A complete proof may require either:
1. A refined measure (e.g., counting the depth of nested deltas in the third argument of `recDelta`)
2. A different termination argument (e.g., lexicographic with three components)
