-- integrate (delta t) --> void: Kappa either drops or ties with mu drop
| _, _, Step.R_int_delta t => by

  · have hkdrop : kappa void < kappa (integrate (delta t)) := by simp [kappa]; omega
    exact drop_left hkdrop
  · simp at h
    have hmu : mu void < mu (integrate (delta t)) := by simp [mu]; omega
    have hke : kappa void = kappa (integrate (delta t)) := by simp [kappa]; omega
    exact drop_right hmu hke

| _, _, Step.R_rec_zero b s => by

    have hke : kappa b = kappa (recDelta b s void) := by simp [kappa]; omega
    exact drop_right hmu hke

| _, _, Step.R_rec_succ b s n => by

    simp [kappa]; omega

| _, _, Step.R_eq_refl a => by

    exact drop_left hkdrop
  · simp at h

-- eqW a b --> integrate (merge a b): Kappa ties, mu drops
| _, _, Step.R_eq_diff a b => by
  have hmu : mu (integrate (merge a b)) < mu (eqW a b) := by simp [mu]; omega
  have hke : kappa (integrate (merge a b)) = kappa (eqW a b) := by simp [kappa]
  exact drop_right hmu hke

**Key patterns:**
- **R_int_delta, R_rec_zero, R_eq_refl** use case analysis: either `kappa` strictly decreases (via `drop_left`) or `kappa` ties and `mu` strictly decreases (via `drop_right`)
- **R_rec_succ** has strict `kappa` decrease only (the delta-wrapping adds 1 to the recursion depth, which is eliminated in the output)
- **R_eq_diff** has tied `kappa` with strict `mu` decrease (4+2·μ → 3+2·μ)
