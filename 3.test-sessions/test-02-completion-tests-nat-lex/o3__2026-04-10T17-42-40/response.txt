For each remaining rule you already have the relevant `simp` lemmas, so all that is left is to combine

-- integrate (delta t) --> void
| _, _, Step.R_int_delta t => by
    -- μ strictly drops; κ may drop or tie (κ void = 0)
    have hμ : mu void < mu (integrate (delta t)) := by
      simp [mu]
    by_cases hκ : kappa void = kappa (integrate (delta t)) with
    | isTrue  h_eq  => exact drop_right hμ h_eq
    | isFalse h_neq =>
        have h_lt : kappa void < kappa (integrate (delta t)) := by
          -- κ void = 0 < κ t  unless κ t = 0 (already in first branch)
          simp [kappa] at h_neq ⊢
        exact drop_left h_lt

-- recDelta b s void --> b
| _, _, Step.R_rec_zero b s => by
    -- μ strictly drops by the constant 3 + μ s
    have hμ : mu b < mu (recDelta b s void) := by
      simp [mu]
    by_cases hκ : kappa b = kappa (recDelta b s void) with
    | isTrue  h_eq  => exact drop_right hμ h_eq
    | isFalse h_neq =>
        have h_lt : kappa b < kappa (recDelta b s void) := by
          -- show κ b < max (κ b) (κ s) using h_neq
          simpa [kappa] using Nat.lt_of_le_of_ne (Nat.le_max_left _ _) h_neq
        exact drop_left h_lt

-- recDelta b s (delta n) --> app s (recDelta b s n)
| _, _, Step.R_rec_succ b s n => by
    -- κ drops by the outer “+1”
    have hκ : kappa (app s (recDelta b s n)) < kappa (recDelta b s (delta n)) := by
      simp [kappa]                       -- unfolds to an inequality of maxima
      exact Nat.lt_succ_of_le (Nat.le_max_right _ _)
    exact drop_left hκ                  -- no μ reasoning necessary

-- eqW a a --> void
| _, _, Step.R_eq_refl a => by
    -- κ either drops (if κ a > 0) or ties; μ always drops
    have hμ : mu void < mu (eqW a a) := by
      simp [mu]
    by_cases hκ : kappa void = kappa (eqW a a) with
    | isTrue  h_eq  => exact drop_right hμ h_eq
    | isFalse h_neq =>
        have h_lt : kappa void < kappa (eqW a a) := by
          simp [kappa] at h_neq ⊢
        exact drop_left h_lt

-- eqW a b --> integrate (merge a b)
| _, _, Step.R_eq_diff a b => by
    -- κ ties (both sides: max κ a κ b); μ drops by exactly 1
    have hμ : mu (integrate (merge a b)) < mu (eqW a b) := by
      simp [mu]
    have hκ : kappa (integrate (merge a b)) = kappa (eqW a b) := by
      simp [kappa]
    exact drop_right hμ hκ

1.  Prove `hμ` (strict decrease of `mu`) with a single `simp`.  
2.  Decide whether `kappa` strictly drops or merely ties via `by_cases`.  
3.  Feed the appropriate facts to `drop_left` or `drop_right`.
