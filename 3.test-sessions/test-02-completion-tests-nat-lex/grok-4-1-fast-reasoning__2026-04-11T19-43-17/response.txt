-- integrate (delta t) --> void : kappa drops to 0 or ties and mu drops by 1 + mu(t)
| _, _, Step.R_int_delta t => by
  have hmu : mu void < mu (integrate (delta t)) := by simp [mu]; omega
  have hle : kappa void ≤ kappa (integrate (delta t)) := by simp [kappa]; linarith
  cases eq_or_lt_of_le hle with
  | inl h_eq => exact drop_right hmu h_eq
  | inr h_lt => exact drop_left h_lt

-- recDelta b s void --> b : kappa(recDelta) = max(kappa b, kappa s) ≥ kappa b,
-- mu drops by 3 + mu s
| _, _, Step.R_rec_zero b s => by
  have hmu : mu b < mu (recDelta b s void) := by simp [mu]; omega
  have hle : kappa b ≤ kappa (recDelta b s void) := by simp [kappa]; linarith
  cases eq_or_lt_of_le hle with
  | inl h_eq => exact drop_right hmu h_eq
  | inr h_lt => exact drop_left h_lt

-- recDelta b s (delta n) --> app s (recDelta b s n) :
-- kappa drops by 1 (rec unwrapping)
| _, _, Step.R_rec_succ b s n => by
  have hkappa : kappa (app s (recDelta b s n)) < kappa (recDelta b s (delta n)) := by
    simp [kappa]
    apply Nat.lt_of_le_of_lt
    · apply Nat.max_le_max <;> simp [kappa]; linarith
    · simp
  exact drop_left hkappa

-- eqW a a --> void : kappa drops to 0 or ties and mu drops by 4 + 2 mu(a)

-- eqW a b --> integrate (merge a b) : kappa ties, mu drops by 1
| _, _, Step.R_eq_diff a b => by
  have hmu : mu (integrate (merge a b)) < mu (eqW a b) := by simp [mu]; omega
  have hk  : kappa (integrate (merge a b)) = kappa (eqW a b) := by simp [kappa]
  exact drop_right hmu hk
